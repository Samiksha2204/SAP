REPORT  Z15PROG12.
DATA  : STRING1 TYPE CHAR50 VALUE 'HELLO',
        STRING2 TYPE CHAR50 VALUE 'ABAP',
        STRING3 TYPE CHAR50 VALUE 'DEVELOPER',
        STRING4 TYPE CHAR100.

CONCATENATE STRING1 STRING2 STRING3 INTO STRING4 SEPARATED BY SPACE. "FOR SPACE IN BETWEEN EACH WORD.
WRITE:/ STRING4.
ULINE.

CONCATENATE STRING1 STRING2 STRING3 INTO STRING4 SEPARATED BY '_'. "FOR UNDERSCORE IN BETWEEN EACH WORD.
WRITE:/ STRING4.
ULINE.

*WRITE :/ 'STRING VALUES BEFORE CLEAR: ',STRING1, STRING2, STRING3.
CLEAR : STRING1, STRING2, STRING3.
*WRITE :/ 'STRING VALUES AFTER CLEAR: ',STRING1, STRING2, STRING3.
*ULINE.

SPLIT STRING4 AT '_' INTO STRING1 STRING2 STRING3.
WRITE :/ 'STRING VALUES AFTER SPLIT: ',STRING1, STRING2, STRING3.
